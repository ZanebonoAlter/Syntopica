// SectorApprovalPanel.vue — replace lines 129-138
const api = useBoardConceptsApi()

async function executeApproval(acceptedIds: string[]) {
  const acceptedSet = new Set(acceptedIds)
  const filteredDiff: SectorDiff = {
    keep: (props.diff.keep || []).filter((_, i) => acceptedSet.has(`keep-${i}`)),
    add: (props.diff.add || []).filter((_, i) => acceptedSet.has(`add-${i}`)),
    merge: (props.diff.merge || []).filter((_, i) => acceptedSet.has(`merge-${i}`)),
    split: (props.diff.split || []).filter((_, i) => acceptedSet.has(`split-${i}`)),
    affected_tag_count: props.diff.affected_tag_count,
  }

  executing.value = true
  execError.value = null
  execProgress.value = { current: 0, total: filteredDiff.add.length + filteredDiff.merge.length + filteredDiff.split.length, label: '准备执行' }

  try {
    const res = await api.confirmRegenerateSectors(props.category, filteredDiff)
    if (res.success) {
      execResult.value = {
        sectorCounts: {
          created: filteredDiff.add.length,
          merged: filteredDiff.merge.length,
          split: filteredDiff.split.length,
        },
        affectedTags: filteredDiff.affected_tag_count,
        success: filteredDiff.add.length + filteredDiff.merge.length + filteredDiff.split.length,
        failed: 0,
        failedReasons: [],
      }
      execProgress.value = { current: execProgress.value.total, total: execProgress.value.total, label: '完成' }
    } else {
      execError.value = res.error || '执行失败'
    }
  } catch (e) {
    execError.value = e instanceof Error ? e.message : '执行失败'
  } finally {
    executing.value = false
  }
}

function handleConfirm() {
